#include<iostream>
#include<cmath>
using namespace std;

const double PI=3.14159;
class Trigonometric {
protected:
    double arc; // 弧度
    double value; // 三角函数值
public:
    Trigonometric(double t) : value(0) 
    {
      if(t>=360)
      while (1)
      {
        t-=360;
        if(t<360) break;
      }
      arc=t * PI / 180 ;
    }
};

class Sine : public Trigonometric {
public:
    Sine(double t) : Trigonometric(t) {} 

    double f1(double x, int n) {
        double result = 1;
        for (int i = 0; i < n; ++i) {
            result *= x;
        }
        return result;
    }

    int f2(int n) {
        if (n == 0 || n == 1) {
            return 1;
        } else {
            return n * f2(n - 1);
        }
    }

    void function() {
        double t = 0;
        double x = arc;
        while (x >= 2 * PI) {
            x -= 2 * PI;
        }

        for (int n = 1, k = 1; ; n += 2, k *= -1) {
            double term = k * f1(x, n) / f2(n);
            if (abs(term) < 1e-6) {
                break;
            }
            t += term;
        }
        value = t;
    }

    void show() {
        cout << "sin(" << arc << ")=" << value << endl;
    }
};


int main()
{
    double d;
    cout<<"请输入度数：";
    cin>>d;
    Sine sin(d);
    sin.function();
    sin.show();
    return 0;
}
